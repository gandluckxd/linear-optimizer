# Настройки headless-оптимизатора москитных сеток.
# Скопируйте этот файл рядом с mos-optimizer-runner.exe
# и переименуйте в linear_optimizer_mos.txt.
#
# Формат каждой настройки: параметр = значение
# Пустые строки и комментарии, начинающиеся с # или ;, пропускаются.
# Меняйте только значения справа от знака =. Имена параметров слева менять нельзя.
# Логические значения: да/нет, true/false или 1/0.
# Дробные числа можно указывать через точку или запятую: 4.5 либо 4,5.

# Адрес HTTP API, через который runner получает данные и записывает результат.
# Обязательный параметр. Обычно API работает на этом же компьютере.
api_url = http://127.0.0.1:8001

# Файл журнала работы runner. Относительный путь считается от этого TXT-файла.
# Папка logs будет создана автоматически.
log_file = logs/mos-optimizer-runner.log

# ---------------------------------------------------------------------------
# ЛИНЕЙНЫЙ РАСКРОЙ ПРОФИЛЯ
# ---------------------------------------------------------------------------

# Ширина пропила пилы, мм. Учитывается между соседними деталями на одном хлысте.
blade_width_mm = 5.0

# Минимальная длина делового остатка, мм.
# Обрезок короче этого значения считается отходом, а не складским остатком.
min_remainder_mm = 300.0

# Максимальный процент отхода для линейного раскроя.
# Параметр сохранён для совместимости с GUI; текущий алгоритм его не применяет.
max_waste_percent = 15.0

# да — стараться получать одинаковые или похожие карты раскроя целых хлыстов;
# нет — не давать таким картам дополнительный приоритет.
pair_optimization = да

# да — использовать складские остатки профиля (они имеют приоритет);
# нет — раскраивать только целые хлысты.
use_remainders = да

# Лимит времени оптимизации, секунд.
# Параметр сохранён для совместимости с GUI; текущий алгоритм его не применяет.
time_limit_seconds = 60

# Технологический отступ от начала хлыста, мм. Недоступен для размещения деталей.
begin_indent_mm = 10.0

# Технологический отступ от конца хлыста, мм. Недоступен для размещения деталей.
end_indent_mm = 10.0

# Минимальный отход, мм, записываемый в карту раскроя Altawin (поле MINTRASH).
# На выбор раскроя в текущем алгоритме не влияет.
min_trash_mm = 50.0

# ---------------------------------------------------------------------------
# ТОНКАЯ НАСТРОЙКА ПАРНОЙ ОПТИМИЗАЦИИ
# Обычно эти четыре значения менять не требуется.
# Большее значение бонуса сильнее влияет на выбор соответствующей карты.
# ---------------------------------------------------------------------------

# Бонус за создание карты, полностью совпадающей с картой другого целого хлыста.
pairing_exact_bonus = 3000.0

# Максимальный бонус за частичное совпадение карт.
# Фактический бонус умножается на долю совпадения.
pairing_partial_bonus = 1000.0

# Минимальная доля совпадения для частичного бонуса: от 0 до 1.
# Например, 0.7 означает совпадение не менее чем на 70 %.
pairing_partial_threshold = 0.7

# Бонус за начало простой карты из одного вида детали на пустом целом хлысте.
pairing_new_simple_bonus = 150.0

# ---------------------------------------------------------------------------
# РАСКРОЙ ПОЛОТНА ФИБЕРГЛАССА
# ---------------------------------------------------------------------------

# Минимальная ширина делового остатка полотна, мм.
# Проверяется после вычета remainder_indent со всех сторон.
planar_min_remainder_width = 500.0

# Минимальная высота делового остатка полотна, мм.
# Остаток можно повернуть: сравниваются его меньшая и большая стороны.
planar_min_remainder_height = 500.0

# Ширина реза полотна, мм.
# Параметр сохранён для совместимости с GUI; текущий алгоритм его не применяет.
planar_cut_width = 1.0

# Отступ от каждой стороны целого листа, мм.
# Параметр сохранён для совместимости с GUI; текущий алгоритм его не применяет.
sheet_indent = 15.0

# Отступ от каждой стороны найденного остатка, мм.
# После вычета этих отступов проверяется минимальный размер делового остатка.
remainder_indent = 15.0

# Максимальный процент отхода при раскрое полотна.
# Параметр сохранён для совместимости с GUI; текущий алгоритм его не применяет.
planar_max_waste_percent = 5.0

# да — разрешить использование складских остатков полотна.
# Параметр передаётся алгоритму, но в текущей версии остатки загружаются всегда.
use_warehouse_remnants = да

# да — разрешить поворот деталей полотна на 90°; нет — сохранить исходную ориентацию.
allow_rotation = да

# ---------------------------------------------------------------------------
# ЗАПИСЬ РЕЗУЛЬТАТА
# --dry-run всегда отключает любые записи, независимо от этих параметров.
# ---------------------------------------------------------------------------

# да — записать рассчитанные карты раскроя в Altawin;
# нет — только рассчитать результат, не выполняя следующие две операции.
save_result = да

# да — после записи карт создать черновые складские документы OUTLAY и SUPPLY.
# Имеет смысл только при save_result = да. Повторный запуск создаст новые документы.
adjust_materials = да

# да — после записи карт присвоить деталям номера производственных ячеек.
# Имеет смысл только при save_result = да.
distribute_cells = да

# да — считать неразмещённые детали профиля ошибкой и ничего не записывать;
# нет — разрешить сохранение неполного линейного раскроя, как в старом GUI.
require_complete_placement = да

# да — при ошибке раскроя полотна ничего не записывать;
# нет — записать линейный раскрой, а ошибку фибергласса оставить предупреждением.
require_fiberglass_success = да
